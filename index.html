# 快速开始

# 安装组件库

npm i ming-currency-ui

# 引用组件库

在main.js中引入组件库

// 全部引入
import 'ming-currency-ui/dist/css/index.css';
import MUI from "ming-currency-ui";
Vue.use(MUI);

// 按需引入
import 'ming-currency-ui/dist/css/card.css';
import { Card } from "ming-currency-ui";
Vue.use(Card);
